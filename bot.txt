Client:

api.controller = function($scope, $timeout) {
  var c = this;
  
  // 1. Initialize State (False means start minimized)
  c.chatOpen = false; 

  c.messages = [
    { text: "Hello! I can help with Password Reset, Duo, or VPN. What do you need?", author: "Bot" }
  ];
  c.userText = "";

  // 2. Add the Toggle Function
  c.toggleChat = function() {
    c.chatOpen = !c.chatOpen;
    
    // Auto-scroll to bottom when opening
    if(c.chatOpen) {
        $timeout(scrollToBottom, 100);
    }
  };

  c.sendMessage = function() {
    if (!c.userText) return;

    // Add User Message
    c.messages.push({ text: c.userText, author: "User" });
    var userMsg = c.userText;
    c.userText = "";
    
    scrollToBottom();

    // API Call to your Azure Bot
    var botURL = "https://YOUR-APP-NAME.azurewebsites.net/chat"; 

    $scope.server.get({
        action: 'chat_proxy',
        message: userMsg,
        endpoint: botURL
    }).then(function(r) {
        c.messages.push({ text: r.data.answer, author: "Bot" });
        scrollToBottom();
    });
  };

  function scrollToBottom() {
    $timeout(function() {
      var chatContainer = document.getElementById("chatContainer");
      if(chatContainer) {
          chatContainer.scrollTop = chatContainer.scrollHeight;
      }
    }, 0);
  }
};
