Client Script:

api.controller = function($http, $scope, $timeout) {
  var c = this;
  
  // UI State
  c.isOpen = false;
  c.isTyping = false;
  c.userMessage = '';
  c.unreadCount = 1;
  c.messages = [];
  c.sessionId = null;

  function getTime() {
    return new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
  }

  function scrollToBottom() {
    $timeout(function() {
      // TARGETING THE NEW ID
      var chatBody = document.getElementById('sn-chat-body');
      if(chatBody) chatBody.scrollTop = chatBody.scrollHeight;
    }, 100);
  }

  c.toggleChat = function() {
    c.isOpen = !c.isOpen;
    if (c.isOpen) {
      c.unreadCount = 0;
      scrollToBottom();
      $timeout(function() { 
          var input = document.getElementById('chatInput');
          if(input) input.focus(); 
      }, 300);
    }
  };

  c.closeChat = function() {
      c.isOpen = false;
  };

  c.checkEnter = function(event) {
    if (event.which === 13) c.sendMessage();
  };

  // Initial Greeting
  $timeout(function() {
    c.messages.push({
      text: "Hello! I am your Virtual Agent. How can I help you?",
      isUser: false,
      time: getTime()
    });
  }, 500);

  c.sendMessage = function() {
    if (!c.userMessage.trim()) return;

    var txt = c.userMessage;
    c.messages.push({ text: txt, isUser: true, time: getTime() });
    c.userMessage = '';
    c.isTyping = true;
    scrollToBottom();

    var payload = {
      message: txt,
      session_id: c.sessionId,
      user_id: c.data.user_sys_id
    };

    // REPLACE WITH YOUR AZURE URL
    var azureUrl = 'https://YOUR-APP-NAME.azurewebsites.net/chat';

    $http.post(azureUrl, payload).then(function(response) {
      c.isTyping = false;
      c.sessionId = response.data.session_id;
      
      c.messages.push({ text: response.data.response, isUser: false, time: getTime() });

      if (response.data.action_taken === 'incident_created') {
        c.messages.push({
          text: "Ticket " + response.data.ticket_number + " has been created.",
          isUser: false,
          time: getTime()
        });
      }
      scrollToBottom();
    }, function(error) {
      c.isTyping = false;
      c.messages.push({ text: "Connection error. Please try again.", isUser: false, time: getTime() });
      scrollToBottom();
    });
  };
};
